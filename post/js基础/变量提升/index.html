变量提升：在当前作用域下，会把带 var 和 function 进行提前声明或定义
注意： 函数提升要比变量提升的优先级要高，且不会被变量声明覆盖，但是会被变量赋值之后覆盖
注意： 块级作用域只对 let/const/function 有效，对 var 无效
注意： 变量没有 var 声明为全局变量，有 var 声明则是 function 的私有变量

// 变量提升var a , function a
console.log(a)  // undefined

var a = 0

if(true){

  console.log(a)  // 块级作用域function a提升
  a = 1
  console.log(a)  // 1
  function a() {} // 把a之前的变化，映射全局a = 1
  console.log(a)  // 1
  a = 21  // 私有，未映射到全局
  console.log(a)  // 21

}

console.log(a)  // 1

// 变量提升
// var n ; var c ; function a() {}

var n = 0

function a() {

  // 私有作用域， var n ,function b() {}
  var n = 10
  function b() {
    n++   // 操作自己上级的作用域，不操作全局
    console.log(n)
  }
  b()
  return b

}

var c = a()
c() // 相当于调用b函数
console.log(n)  // 全局变量不受影响，function a操作的是 `var n = 10` 这里的私有变量

var foo = 1

function bar() {

  if(!foo) {
    var foo = 10
  }
  console.log(foo)

}

bar()

// 打印结果
// 10

分析：私有作用域中的 if 块级作用域中 var foo 变量提升，此时的foo为 undefined ，!foo为true，符合条件 foo = 10 ,结果打印 10
